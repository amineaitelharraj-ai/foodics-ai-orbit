  const CHATS_PER_PAGE = 3

  function ChatHistorySection({ date, chats, onSelectChat, onDeleteChat }: Props) {
    const [visibleCount, setVisibleCount] = useState(CHATS_PER_PAGE)

    const visibleChats = chats.slice(0, visibleCount)
    const hasMore = visibleCount < chats.length
    const remainingCount = chats.length - visibleCount
    const isExpanded = visibleCount > CHATS_PER_PAGE

    const handleLoadMore = () => {
      setVisibleCount(prev => Math.min(prev + CHATS_PER_PAGE, chats.length))
    }

    const handleShowLess = () => {
      setVisibleCount(CHATS_PER_PAGE)
    }

    return (
      <div className="mb-4">
        {/* Date header */}
        <div className="flex items-center gap-2 text-sm text-gray-500 mb-2">
          <CalendarIcon className="w-4 h-4" />
          <span>{date}</span>
        </div>

        {/* Chat list */}
        <div className="space-y-1">
          {visibleChats.map(chat => (
            <ChatItem key={chat.id} chat={chat} ... />
          ))}
        </div>

        {/* Buttons */}
        <div className="flex gap-2 mt-2">
          {hasMore && (
